def add_comment(request, pk):
    """添加评论"""
    form = CommentForm(request.POST)
    if form.is_valid():
        comment = form.save(commit=False)
        comment.post_id = pk
        comment.save()
        return redirect('blog.views.post_detail', pk=pk)
